### 方案 2：Python 脚本自动抓取

```python
#!/usr/bin/env python3
# filename: /home/admin/.openclaw/workspace/scripts/ai-news-daily.py

import feedparser
import requests
from datetime import datetime
import json

# RSS 源列表
RSS_FEEDS = {
    "OpenAI": "https://openai.com/news/rss",
    "Anthropic": "https://www.anthropic.com/news/rss",
    "Google AI": "https://blog.google/technology/ai/rss",
    "Meta AI": "https://ai.meta.com/blog/rss",
    "Microsoft AI": "https://blogs.microsoft.com/ai/rss",
    "DeepMind": "https://deepmind.google/discover/blog/rss",
    "The Batch": "https://www.deeplearning.ai/the-batch/rss",
}

def fetch_daily_news():
    """抓取过去 24 小时的 AI 新闻"""
    all_news = []
    
    for source, url in RSS_FEEDS.items():
        try:
            feed = feedparser.parse(url)
            for entry in feed.entries[:5]:  # 每个源取最新 5 条
                news_item = {
                    "source": source,
                    "title": entry.title,
                    "link": entry.link,
                    "published": entry.get("published", ""),
                    "summary": entry.get("summary", "")[:200]  # 摘要前 200 字
                }
                all_news.append(news_item)
        except Exception as e:
            print(f"Error fetching {source}: {e}")
    
    # 按时间排序
    all_news.sort(key=lambda x: x["published"], reverse=True)
    
    return all_news

def generate_markdown(news_list):
    """生成 Markdown 格式的每日总结"""
    md = f"# 📰 AI 领域每日动态\n\n"
    md += f"**📅 日期**: {datetime.now().strftime('%Y-%m-%d')}\n\n"
    md += f"**📊 总计**: {len(news_list)} 条新闻\n\n"
    md += "---\n\n"
    
    # 按来源分组
    by_source = {}
    for news in news_list:
        source = news["source"]
        if source not in by_source:
            by_source[source] = []
        by_source[source].append(news)
    
    for source, source_news in by_source.items():
        md += f"## {source}\n\n"
        for i, news in enumerate(source_news[:5], 1):
            md += f"### {i}. {news['title']}\n\n"
            md += f"- 📅 {news['published']}\n"
            md += f"- 🔗 [阅读原文]({news['link']})\n"
            if news['summary']:
                md += f"- 📝 {news['summary']}...\n"
            md += "\n"
        md += "---\n\n"
    
    return md

if __name__ == "__main__":
    # 抓取新闻
    news = fetch_daily_news()
    
    # 生成 Markdown
    md_content = generate_markdown(news)
    
    # 保存到文件
    timestamp = datetime.now().strftime("%Y%m%d")
    md_file = f"/home/admin/.openclaw/workspace/AI-NEWS-DAILY-{timestamp}.md"
    
    with open(md_file, 'w', encoding='utf-8') as f:
        f.write(md_content)
    
    print(f"Saved {len(news)} news items to {md_file}")
```

### 方案 3：直接网页抓取（备选）

如果 RSS 不可用，直接抓取官网：

```python
import requests
from bs4 import BeautifulSoup

def fetch_openai_news():
    """抓取 OpenAI 最新新闻"""
    url = "https://openai.com/news"
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    
    articles = []
    for article in soup.select('article')[:10]:
        title = article.select_one('h3')
        link = article.select_one('a')['href']
        date = article.select_one('time')
        
        articles.append({
            "title": title.text if title else "N/A",
            "link": f"https://openai.com{link}" if link else "N/A",
            "date": date.text if date else "N/A"
        })
    
    return articles
```

```python
def upload_to_getnote(md_content, date_str):
    """上传到 Get 笔记"""
    import os
    
    API_KEY = os.getenv('GETNOTE_API_KEY')
    CLIENT_ID = os.getenv('GETNOTE_CLIENT_ID')
    
    headers = {
        'Authorization': API_KEY,
        'X-Client-ID': CLIENT_ID,
        'Content-Type': 'application/json'
    }
    
    data = {
        'title': f'📰 AI 领域每日动态 - {date_str}',
        'content': md_content,
        'note_type': 'plain_text',
        'tags': ['AI 动态', '每日总结', '行业资讯']
    }
    
    response = requests.post(
        'https://openapi.biji.com/open/api/v1/resource/note/save',
        headers=headers,
        json=data
    )
    
    result = response.json()
    if result.get('success'):
        note_id = result['data']['note_id']
        print(f"Uploaded to Get 笔记：{note_id}")
        
        # 添加到 Ai &具身智能知识库
        requests.post(
            'https://openapi.biji.com/open/api/v1/resource/knowledge/note/batch-add',
            headers=headers,
            json={'topic_id': 'oYpEp190', 'note_ids': [note_id]}
        )
```
